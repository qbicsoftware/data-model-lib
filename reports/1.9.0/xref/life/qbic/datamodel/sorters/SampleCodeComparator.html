26  /**
27   * Compares IBarcodeBeans by sample ID
28   * 
29   * @author Andreas Friedrich

31   */
32  public class SampleCodeComparator implements Comparator<IBarcodeBean> {
33    
34    private static final Logger logger = LogManager.getLogger(SampleCodeComparator.class);
35  
36    private static final SampleCodeComparatorator.html#SampleCodeComparator">SampleCodeComparator instance = new SampleCodeComparator();
37  
38    public static SampleCodeComparator getInstance() {
39      return instance;
40    }
41  
42    private SampleCodeComparator() {}
43  
44    @Override
45    public int compare(IBarcodeBean../../../../life/qbic/datamodel/printing/IBarcodeBean.html#IBarcodeBean">IBarcodeBean o1, IBarcodeBean o2) {
46      String c1 = o1.getCode();
47      String c2 = o2.getCode();
48      if (!c1.startsWith("Q") || c1.contains("ENTITY") || !c2.startsWith("Q")
49          || c2.contains("ENTITY"))
50        return o1.getCode().compareTo(o2.getCode());
51      try {
52        // compares sample codes by projects, ending letters (999A --> 001B) and numbers (001A -->
53        // 002A)
54        int projCompare = c1.substring(0, 5).compareTo(c2.substring(0, 5));
55        int numCompare = c1.substring(5, 8).compareTo(c2.substring(5, 8));
56        int letterCompare = c1.substring(8, 9).compareTo(c2.substring(8, 9));
57        if (projCompare != 0)
58          return projCompare;
59        else {
60          if (letterCompare != 0)
61            return letterCompare;
62          else
63            return numCompare;
64        }
65      } catch (Exception e) {
66        logger.warn("Could not split code " + c1 + " or " + c2
67            + ". Falling back to primitive lexicographical comparison.");
68      }
69      return o1.getCode().compareTo(o2.getCode());
70    }
71  }
