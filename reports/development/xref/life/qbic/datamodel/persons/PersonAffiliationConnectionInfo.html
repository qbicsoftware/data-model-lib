18  package life.qbic.datamodel.persons;
19  
20  public class PersonAffiliationConnectionInfo {
21    
22    private int person_id;
23    private int affiliation_id;
24    private String role;
25    
26    public PersonAffiliationConnectionInfo(int person_id, int affiliation_id, String role) {
27      super();
28      this.person_id = person_id;
29      this.affiliation_id = affiliation_id;
30      this.role = role;
31    }
32    
33    public int getPerson_id() {
34      return person_id;
35    }
36    public int getAffiliation_id() {
37      return affiliation_id;
38    }
39    public String getRole() {
40      return role;
41    }
42  
43  }
